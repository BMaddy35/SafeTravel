       // Here you can use any suitable way to get element. Tag/class/id/etc.
       var messenger = document.querySelector('df-messenger');
 
       //These will render custom text message and custom card, as if they were served from Dialogflow as simple text response and card response
       messenger.renderCustomText('Hello, this is Test bot for static welcome. What can I help you with?');
       messenger.renderCustomCard([{ type: 'info', title: 'Some title', subtitle: 'some subtitle' }, { type: 'description', text: ['test'] }]);
 
       // This will show a mini version of messages list with the very first message
       messenger.showMinChat();
   });
</script>
